CREATE TABLE customer (
  ID INTEGER,
  CUSTOMER_NAME VARCHAR(255),
  CYTI VARCHAR(255),
  PRIMARY KEY (ID)
);

CREATE TABLE products (
  ID INTEGER,
  CUSTOMER_ID INTEGER,
  PRODUCTS_NAME VARCHAR(255),
  AMOUNT INTEGER,
  COST INTEGER,
  PRIMARY KEY (ID),
  FOREIGN KEY (CUSTOMER_ID) REFERENCES customer (ID)
);
